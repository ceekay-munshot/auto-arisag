CKAN Data API

Få tillgång till resursens data via ett webb-API med kraftfullt frågestöd.. Further information in the main CKAN Data API and DataStore documentation.

Code examples:

Get 5 results containing "jones" in any field: 
curl https://ckandev.indiadataportal.com/sv/api/action/datastore_search \
  -H"Authorization:$API_TOKEN" -d '
{
  "resource_id": "cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a",
  "limit": 5,
  "q": "jones"
}'
const resp = await fetch(`https://ckandev.indiadataportal.com/sv/api/action/datastore_search`, {
    method: 'POST',
    headers: {
        'content-type': 'application/json',
        authorization: API_TOKEN
    },
    body: JSON.stringify({
        resource_id: 'cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a',
        limit: 5,
        q: 'jones'
    })
})
await resp.json()
$json = @'
{
  "resource_id": "cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a",
  "limit": 5,
  "q": "jones"
}
'@
$response = Invoke-RestMethod https://ckandev.indiadataportal.com/sv/api/action/datastore_search`
  -Method Post -Body $json -Headers @{"Authorization"="$API_TOKEN"}
$response.result.records

(using the ckanapi client library) 

from ckanapi import RemoteCKAN

rc = RemoteCKAN('https://ckandev.indiadataportal.com/sv/', apikey=API_TOKEN)
result = rc.action.datastore_search(
    resource_id="cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a",
    limit=5,
    q="jones",
)
print(result['records'])
library(httr2)

req <- request("https://ckandev.indiadataportal.com/sv/api/action/datastore_search")
result <- req %>% 
    req_headers(Authorization = API_TOKEN) %>% 
    req_body_json(list(
        resource_id = 'cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a',
        limit = 5,
        q = 'jones'))
    req_perform %>% 
    resp_body_json
Get results with either "watershed" or "survey" as subject and "active" as its stage: 
curl https://ckandev.indiadataportal.com/sv/api/action/datastore_search \
-H"Authorization:$API_TOKEN" -d '
{
"resource_id": "cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a",
  "filters": {
    "subject": ["watershed", "survey"],
    "stage": "active"
  }
}'
const resp = await fetch(`https://ckandev.indiadataportal.com/sv/api/action/datastore_search`, {
    method: 'POST',
    headers: {
        'content-type': 'application/json',
        authorization: API_TOKEN
    },
    body: JSON.stringify({resource_id: 'cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a', filters: {
        subject: ['watershed', 'survey'],
        stage: 'active'
    }})})
await resp.json()
$json = @'
{
  "resource_id": "cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a",
  "filters": {
    "subject": ["watershed", "survey"],
    "stage": "active"
  }
}
'@
$response = Invoke-RestMethod https://ckandev.indiadataportal.com/sv/api/action/datastore_search`
  -Method Post -Body $json -Headers @{"Authorization"="$API_TOKEN"}
$response.result.records
from ckanapi import RemoteCKAN

rc = RemoteCKAN('https://ckandev.indiadataportal.com/sv/', apikey=API_TOKEN)
result = rc.action.datastore_search(
    resource_id="cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a",
    filters={
      "subject": ["watershed", "survey"],
      "stage": "active",
    },
)
print(result['records'])
library(httr2)

req <- request("https://ckandev.indiadataportal.com/sv/api/action/datastore_search")
result <- req %>% 
    req_headers(Authorization = API_TOKEN) %>% 
    req_body_json(list(
        resource_id='cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a', 
        filters = list(
            subject = list("watershed", "survey"), 
            stage = "active")))
    req_perform %>% 
    resp_body_json
SQL query example: 
curl https://ckandev.indiadataportal.com/sv/api/action/datastore_search_sql \
  -H"Authorization:$API_TOKEN" -d @- <<END
{
  "sql": "SELECT * FROM \"cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a\" WHERE title LIKE 'jones'"
}
END
const resp = await fetch(`https://ckandev.indiadataportal.com/sv/api/action/datastore_search_sql`, {
    method: 'POST',
    headers: {
        'content-type': 'application/json',
        authorization: API_TOKEN
    },
    body: JSON.stringify({
        sql: `SELECT * FROM "cc32d3e2-7ea3-4b6b-94ab-85e57f6a0a3a" WHERE title LIKE 'jones'`
    })
})
await resp.json()
$json = @'
{
